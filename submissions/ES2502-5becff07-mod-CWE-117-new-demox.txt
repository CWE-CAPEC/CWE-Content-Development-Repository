LANGUAGE: PHP

<?php

function sanitize_for_log(string $input): string {
    // Remove all control characters including newlines, carriage returns, tabs, etc.
    $sanitized = preg_replace('/[\x00-\x1F\x7F]/', '', $input);
    // Trim leading/trailing whitespace
    return trim($sanitized);
}

function validate_username(string $input): string {
    // Sanitize first to strip any injected control characters
    $clean = sanitize_for_log($input);

    // Enforce max length
    if (strlen($clean) > 20) {
        throw new LengthException("Username too long");
    }

    // Allowlist: only alphanumeric characters and underscores
    if (!preg_match('/^[a-zA-Z0-9_]+$/', $clean)) {
        throw new InvalidArgumentException("Username contains invalid characters");
    }

    return $clean;
}

$raw_username = $_GET['username'] ?? '';

try {
    $username = validate_username($raw_username);

} catch (LengthException $e) {
    $safe_log = sanitize_for_log($raw_username);
    error_log("User registration failed (too long): " . substr($safe_log, 0, 50));

} catch (InvalidArgumentException $e) {
    $safe_log = sanitize_for_log($raw_username);
    error_log("User registration failed (invalid chars): " . substr($safe_log, 0, 50));
}
?>
